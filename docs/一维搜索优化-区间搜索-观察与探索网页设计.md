观察与探索网页设计（observe_and_explore.html）

## 文件结构
```
range_search/
├── index.html              # 主入口页面
├── application_and_extension.html # 应用与延伸页面
├── compare_and_insight.html # 对比与洞察页面

├── observe_and_explore.html # 观察与探索页面（目标文件）
├── styles/
│   └── main.css           # 主样式文件
├── scripts/
│   ├── d3.v7.min.js       # D3.js库
│   ├── function-utils.js  # 函数计算工具

│   ├── fibonacci-search.js # 斐波那契搜索算法
│   └── visualization.js   # 可视化引擎
└── data/

```

        const AppState = {
            currentAlgorithm: 'golden-section',

            },
            history: []
        };

            initVisualization();

        // 初始化可视化
        function initVisualization() {
            // 创建SVG画布
            const width = document.getElementById('function-canvas').clientWidth;
            const height = 500;
            
            const svg = d3.select('#function-canvas')
                .append('svg')
                .attr('width', width)
                .attr('height', height)
                .attr('class', 'main-svg');
            
            // 创建绘图区域
            const margin = { top: 40, right: 40, bottom: 60, left: 60 };
            const plotWidth = width - margin.left - margin.right;
            const plotHeight = height - margin.top - margin.bottom;
            
            const plot = svg.append('g')
                .attr('transform', `translate(${margin.left},${margin.top})`);
            
            // 保存绘图上下文
            window.plotContext = {
                svg, plot, width, height,
                margin, plotWidth, plotHeight
            };
            
            // 绘制初始函数
            drawFunction();
        }
        
        // 绘制函数图像
        function drawFunction() {
            const { plot, plotWidth, plotHeight } = window.plotContext;
            
            // 清除之前的绘图
            plot.selectAll('*').remove();
            
            // 创建比例尺
            const xScale = d3.scaleLinear()
                .domain([-5, 5])
                .range([0, plotWidth]);
                
            const yScale = d3.scaleLinear()
                .domain([-3, 10])
                .range([plotHeight, 0]);
            
            // 创建坐标轴
            const xAxis = d3.axisBottom(xScale);
            const yAxis = d3.axisLeft(yScale);
            
            plot.append('g')
                .attr('transform', `translate(0,${plotHeight})`)
                .call(xAxis);
                
            plot.append('g')
                .call(yAxis);
            
            // 绘制网格
            plot.append('g')
                .attr('class', 'grid')
                .call(d3.axisBottom(xScale).tickSize(-plotHeight).tickFormat(''));
                
            plot.append('g')
                .attr('class', 'grid')
                .call(d3.axisLeft(yScale).tickSize(-plotWidth).tickFormat(''));
            
            // 生成函数数据
            const data = [];
            for (let x = -5; x <= 5; x += 0.01) {
                const y = calculateFunction(x, AppState.functionType);
                data.push({ x, y });
            }
            
            // 绘制函数曲线
            const line = d3.line()
                .x(d => xScale(d.x))
                .y(d => yScale(d.y))
                .curve(d3.curveMonotoneX);
            
            plot.append('path')
                .datum(data)
                .attr('class', 'function-line')
                .attr('d', line)
                .attr('fill', 'none')
                .attr('stroke', '#3498db')
                .attr('stroke-width', 2);
            
            // 添加坐标轴标签
            plot.append('text')
                .attr('x', plotWidth / 2)
                .attr('y', plotHeight + 40)
                .attr('text-anchor', 'middle')
                .text('x');
                
            plot.append('text')
                .attr('transform', 'rotate(-90)')
                .attr('x', -plotHeight / 2)
                .attr('y', -40)
                .attr('text-anchor', 'middle')
                .text('f(x)');
            
            // 保存比例尺
            window.plotContext.xScale = xScale;
            window.plotContext.yScale = yScale;
            
            // 绘制当前区间
            drawCurrentInterval();
        }
        
        // 绘制当前区间
        function drawCurrentInterval() {
            const { plot, plotWidth, plotHeight, xScale, yScale } = window.plotContext;
            
            // 清除之前的区间绘制
            plot.selectAll('.interval-visualization').remove();
            
            const { a, b } = AppState.parameters;
            
            // 绘制区间背景
            plot.append('rect')
                .attr('class', 'interval-visualization')
                .attr('x', xScale(Math.min(a, b)))
                .attr('y', 0)
                .attr('width', Math.abs(xScale(b) - xScale(a)))
                .attr('height', plotHeight)
                .attr('fill', 'rgba(52, 152, 219, 0.1)')
                .attr('stroke', 'rgba(52, 152, 219, 0.3)')
                .attr('stroke-width', 1);
            
            // 绘制区间端点
            plot.append('line')
                .attr('class', 'interval-visualization')
                .attr('x1', xScale(a))
                .attr('x2', xScale(a))
                .attr('y1', 0)
                .attr('y2', plotHeight)
                .attr('stroke', '#e74c3c')
                .attr('stroke-width', 2)
                .attr('stroke-dasharray', '5,5');
                
            plot.append('line')
                .attr('class', 'interval-visualization')
                .attr('x1', xScale(b))
                .attr('x2', xScale(b))
                .attr('y1', 0)
                .attr('y2', plotHeight)
                .attr('stroke', '#e74c3c')
                .attr('stroke-width', 2)
                .attr('stroke-dasharray', '5,5');
            
            // 添加端点标签
            plot.append('text')
                .attr('class', 'interval-visualization')
                .attr('x', xScale(a))
                .attr('y', -10)
                .attr('text-anchor', 'middle')
                .text(`a=${a.toFixed(3)}`);
                
            plot.append('text')
                .attr('class', 'interval-visualization')
                .attr('x', xScale(b))
                .attr('y', -10)
                .attr('text-anchor', 'middle')
                .text(`b=${b.toFixed(3)}`);
            
            // 绘制试探点（如果已计算）
            if (AppState.currentIteration > 0 && AppState.history.length > 0) {
                const lastIter = AppState.history[AppState.history.length - 1];
                drawTrialPoints(lastIter);
            }
        }
        
        // 绘制试探点
        function drawTrialPoints(iterationData) {
            const { plot, xScale, yScale } = window.plotContext;
            const { a_try, b_try } = iterationData;
            
            // 清除之前的试探点
            plot.selectAll('.trial-point').remove();
            
            // 绘制试探点垂直线
            plot.append('line')
                .attr('class', 'trial-point')
                .attr('x1', xScale(a_try))
                .attr('x2', xScale(a_try))
                .attr('y1', 0)
                .attr('y2', yScale(calculateFunction(a_try, AppState.functionType)))
                .attr('stroke', '#2ecc71')
                .attr('stroke-width', 1.5)
                .attr('stroke-dasharray', '3,3');
                
            plot.append('line')
                .attr('class', 'trial-point')
                .attr('x1', xScale(b_try))
                .attr('x2', xScale(b_try))
                .attr('y1', 0)
                .attr('y2', yScale(calculateFunction(b_try, AppState.functionType)))
                .attr('stroke', '#2ecc71')
                .attr('stroke-width', 1.5)
                .attr('stroke-dasharray', '3,3');
            
            // 绘制试探点标记
            plot.append('circle')
                .attr('class', 'trial-point')
                .attr('cx', xScale(a_try))
                .attr('cy', yScale(calculateFunction(a_try, AppState.functionType)))
                .attr('r', 6)
                .attr('fill', '#2ecc71');

function calculateFunction(x, functionType) {
    const func = FunctionLibrary[functionType];
    if (!func) {
        console.warn(`未知函数类型: ${functionType}，使用默认二次函数`);
        return FunctionLibrary.quadratic.calculate(x);
    }

// 生成函数数据点
function generateFunctionData(functionType, start, end, step = 0.01) {
    const data = [];
    for (let x = start; x <= end; x += step) {
        data.push({
            x: x,
            y: calculateFunction(x, functionType)
        });
    }
    return data;
}

// 检查函数在区间[a,b]上是否为单谷函数
function isUnimodal(functionType, a, b, samples = 100) {
    const step = (b - a) / samples;
    let decreasing = true;
    let hasValley = false;
    let prevY = calculateFunction(a, functionType);
    
    for (let i = 1; i <= samples; i++) {
        const x = a + i * step;
        const y = calculateFunction(x, functionType);
        
        if (decreasing && y > prevY) {
            decreasing = false;
            hasValley = true;
        } else if (!decreasing && y < prevY) {
            // 出现第二个谷底
            return false;
        }
        
        prevY = y;
    }
    
    return hasValley;
}

class GoldenSectionSearch {
    constructor(func, a, b, epsilon = 0.001, maxIterations = 100) {
        this.func = func;
        this.a = a;
        this.b = b;
        this.epsilon = epsilon;
        this.maxIterations = maxIterations;
        
        // 黄金分割比例
        this.rho = (Math.sqrt(5) - 1) / 2; // ≈ 0.618
        
        // 算法状态
        this.currentIteration = 0;
        this.history = [];
        this.isComplete = false;
        this.result = null;
    }
    
    // 执行一次迭代
    iterate() {
        if (this.isComplete || this.currentIteration >= this.maxIterations) {
            return false;
        }
        
        const l = this.b - this.a;
        
        // 检查停止条件
        if (Math.abs(l) < this.epsilon) {
            this.isComplete = true;
            this.result = (this.a + this.b) / 2;
            return false;
        }
        
        // 计算试探点
        const a_try = this.a + this.rho * l;
        const b_try = this.b - this.rho * l;
        
        // 计算函数值
        const fa = this.func(a_try);
        const fb = this.func(b_try);
        
        // 记录历史
        this.history.push({
            iteration: this.currentIteration + 1,
            a: this.a,
            b: this.b,
            a_try: a_try,
            b_try: b_try,
            fa: fa,
            fb: fb,
            r: this.rho,
            interval_length: l,
            decision: fa > fb ? '舍弃左区间' : '舍弃右区间'
        });
        
        // 更新区间
        if (fa > fb) {
            this.a = a_try; // 舍弃左区间
        } else {
            this.b = b_try; // 舍弃右区间
        }
        
        this.currentIteration++;
        
        return true;
    }
    
    // 执行完整搜索
    search() {
        while (this.iterate()) {
            // 继续迭代直到完成
        }
        
        if (!this.result) {
            this.result = (this.a + this.b) / 2;
        }
        
        return {
            solution: this.result,
            value: this.func(this.result),
            iterations: this.currentIteration,
            finalInterval: [this.a, this.b],
            history: this.history
        };
    }
    
    // 获取当前状态
    getState() {
        return {
            iteration: this.currentIteration,
            a: this.a,
            b: this.b,
            intervalLength: Math.abs(this.b - this.a),
            history: [...this.history]
        };
    }
    
    // 重置算法
    reset() {
        // 注意：需要外部提供初始a,b值
        this.currentIteration = 0;
        this.history = [];
        this.isComplete = false;
        this.result = null;
    }
}

class FibonacciSearch {
    constructor(func, a, b, epsilon = 0.001, maxIterations = 20) {
        this.func = func;
        this.a = a;
        this.b = b;
        this.epsilon = epsilon;
        this.maxIterations = maxIterations;
        
        // 生成斐波那契数列
        this.F = this.generateFibonacci(maxIterations + 2);
        
        // 算法状态
        this.currentIteration = 0;
        this.history = [];
        this.isComplete = false;
        this.result = null;
    }
    
    // 生成斐波那契数列
    generateFibonacci(n) {
        const F = [1, 1];
        for (let i = 2; i <= n; i++) {
            F[i] = F[i - 1] + F[i - 2];
        }
        return F;
    }
    
    // 计算当前迭代的分割比例
    getCurrentRatio() {
        const n = this.maxIterations;
        const k = this.currentIteration;
        if (k >= n - 1) return 0.5; // 最后一次迭代使用中点
        
        return this.F[n - k - 1] / this.F[n - k + 1];
    }
    
    // 执行一次迭代
    iterate() {
        if (this.isComplete || this.currentIteration >= this.maxIterations) {
            return false;
        }
        
        const l = this.b - this.a;
        
        // 检查停止条件
        if (Math.abs(l) < this.epsilon || this.currentIteration === this.maxIterations - 1) {
            this.isComplete = true;
            this.result = (this.a + this.b) / 2;
            return false;
        }
        
        // 计算当前分割比例
        const rho = this.getCurrentRatio();
        
        // 计算试探点
        const a_try = this.a + rho * l;
        const b_try = this.b - rho * l;
        
        // 计算函数值
        const fa = this.func(a_try);
        const fb = this.func(b_try);
        
        // 记录历史
        this.history.push({
            iteration: this.currentIteration + 1,
            a: this.a,
            b: this.b,
            a_try: a_try,
            b_try: b_try,
            fa: fa,
            fb: fb,
            r: rho,
            F_ratio: `${this.F[this.maxIterations - this.currentIteration - 1]}/${this.F[this.maxIterations - this.currentIteration + 1]}`,
            interval_length: l,
            decision: fa > fb ? '舍弃左区间' : '舍弃右区间'
        });
        
        // 更新区间
        if (fa > fb) {
            this.a = a_try; // 舍弃左区间
        } else {
            this.b = b_try; // 舍弃右区间
        }
        
        this.currentIteration++;
        
        return true;
    }
    
    // 执行完整搜索
    search() {
        while (this.iterate()) {
            // 继续迭代直到完成
        }
        
        if (!this.result) {
            this.result = (this.a + this.b) / 2;
        }
        
        return {
            solution: this.result,
            value: this.func(this.result),
            iterations: this.currentIteration,
            finalInterval: [this.a, this.b],
            history: this.history,
            fibonacciSequence: this.F.slice(0, this.maxIterations + 1)
        };
    }
    
    // 获取当前状态
    getState() {
        return {
            iteration: this.currentIteration,
            a: this.a,
            b: this.b,
            intervalLength: Math.abs(this.b - this.a),
            currentRatio: this.getCurrentRatio(),
            history: [...this.history]
        };
    }
    
    // 计算最优迭代次数
    static calculateOptimalIterations(initialLength, epsilon) {
        let n = 2;
        let F = [1, 1];
        
        while (F[F.length - 1] < initialLength / epsilon) {
            F.push(F[F.length - 1] + F[F.length - 2]);
            n++;
        }
        
        return n - 2; // 减去初始的F[0], F[1]
    }
    
    // 重置算法
    reset() {
        this.currentIteration = 0;
        this.history = [];
        this.isComplete = false;
        this.result = null;
    }
}
