\textbf{输入}：目标函数 $f$，初始点 $x_0$，最大迭代次数 $N$

\textbf{输出}：近似极小值点 $x$

\begin{enumerate}
    \item 令 $k = 0$
    \item 计算当前点的一阶导数 $f'(x_k)$ 和二阶导数 $f''(x_k)$
    \begin{itemize}
        \item 若 $k = N$ 或 $f'(x_k) = 0$，输出 $x = x_k$，算法结束
    \end{itemize}
    \item 更新迭代点：
    \begin{itemize}
        \item $x_{k+1} = x_k - \dfrac{f'(x_k)}{f''(x_k)}$
    \end{itemize}
    \item 令 $k = k + 1$，转步骤 2
\end{enumerate}
